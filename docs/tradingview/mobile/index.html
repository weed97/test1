f_zoneTouches(float level, float halfH, int lb) =>
    int touches = 0
    float touchVol = 0.0
    int bounces = 0
    bool broken = false
    for i = 1 to lb
        if bar_index >= i
            float zTop = level + halfH
            float zBot = level - halfH
            bool touched = high[i] >= zBot and low[i] <= zTop
            if touched
                touches += 1
                touchVol += volume[i]
                bool bouncedUp = false
                bool bouncedDn = false
                for j = 0 to 2
                    if i - j >= 0
                        if close[i - j] > zTop
                            bouncedUp := true
                        if close[i - j] < zBot
                            bouncedDn := true
                if bouncedUp or bouncedDn
                    bounces += 1
            if close[i] > zTop + halfH or close[i] < zBot - halfH
                broken := true
    float strength = touches > 0 ? math.min(100, int(touches * 15 + touchVol / math.max(volMa, 1) * 5)) : 0
    [touches, bounces, broken, strength]

f_calcVP() =>
    float hi = ta.highest(high, vpLookback)
    float lo = ta.lowest(low, vpLookback)
    float step = (hi - lo) / vpBins
    float[] bins = array.new_float(vpBins, 0.0)
    float totalVol = 0.0
    for i = 0 to vpLookback - 1
        if bar_index >= i
            float c = close[i]
            float v = volume[i]
            int idx = step > 0 ? int(math.floor((c - lo) / step)) : 0
            idx := math.min(math.max(idx, 0), vpBins - 1)
            array.set(bins, idx, array.get(bins, idx) + v)
            totalVol += v
    int pocIdx = 0
    float maxV = 0.0
    for j = 0 to vpBins - 1
        float bv = array.get(bins, j)
        if bv > maxV
            maxV := bv
            pocIdx := j
    float pocPrice = lo + step * (pocIdx + 0.5)
    float vaVol = array.get(bins, pocIdx)
    int loIdx = pocIdx
    int hiIdx = pocIdx
    float target = totalVol * valueAreaPct / 100.0
    while vaVol < target and (loIdx > 0 or hiIdx < vpBins - 1)
        float volBelow = loIdx > 0 ? array.get(bins, loIdx - 1) : 0.0
        float volAbove = hiIdx < vpBins - 1 ? array.get(bins, hiIdx + 1) : 0.0
        if volAbove >= volBelow and hiIdx < vpBins - 1
            hiIdx += 1
            vaVol += array.get(bins, hiIdx)
        else if loIdx > 0
            loIdx -= 1
            vaVol += array.get(bins, loIdx)
        else if hiIdx < vpBins - 1
            hiIdx += 1
            vaVol += array.get(bins, hiIdx)
        else
            break
    float vah = lo + step * (hiIdx + 1)
    float val = lo + step * loIdx
    float pocStrength = totalVol > 0 ? maxV / totalVol * 100.0 : 0.0
    int[] hvnIdx = array.new_int(3, -1)
    for rank = 0 to 2
        float best = 0.0
        int bestI = -1
        for j = 0 to vpBins - 1
            if j != pocIdx
                float bv = array.get(bins, j)
                bool already = false
                for r = 0 to rank - 1
                    if array.get(hvnIdx, r) == j
                        already := true
                if not already and bv > best
                    best := bv
                    bestI := j
        array.set(hvnIdx, rank, bestI)
    int[] lvnIdx = array.new_int()
    for j = 0 to vpBins - 1
        if j != pocIdx and array.get(bins, j) < maxV * 0.2
            array.push(lvnIdx, j)
    [pocPrice, vah, val, pocStrength, step, lo, bins, hvnIdx, lvnIdx, pocIdx]

f_buildZones() =>
    SrZone[] zones = array.new<SrZone>()
    float clusterR = atr14 * clusterAtrMult
    int pSize = array.size(pivotPrices)
    bool[] used = array.new_bool(pSize, false)
    for i = 0 to pSize - 1
        if not array.get(used, i)
            float pc = array.get(pivotPrices, i)
            bool isH = array.get(pivotIsHigh, i)
            float sumP = pc
            int cnt = 1
            int hiCnt = isH ? 1 : 0
            int loCnt = isH ? 0 : 1
            array.set(used, i, true)
            for j = i + 1 to pSize - 1
                if not array.get(used, j)
                    float pj = array.get(pivotPrices, j)
                    if math.abs(pj - pc) <= clusterR
                        sumP += pj
                        cnt += 1
                        if array.get(pivotIsHigh, j)
                            hiCnt += 1
                        else
                            loCnt += 1
                        array.set(used, j, true)
            float center = sumP / cnt
            float halfZ = atr14 * 0.3
            int tch = 0
            int bull = 0
            int bear = 0
            float tVol = 0.0
            for k = 0 to srLookback - 1
                if bar_index >= k
                    if high[k] >= center - halfZ and low[k] <= center + halfZ
                        tch += 1
                        tVol += volume[k]
                        bool bUp = false
                        bool bDn = false
                        for m = 0 to 2
                            if k - m >= 0
                                if close[k - m] > center + halfZ
                                    bUp := true
                                if close[k - m] < center - halfZ
                                    bDn := true
                        if bUp
                            bull += 1
                        if bDn
                            bear += 1
            float avgV = tch > 0 ? tVol / tch : 0.0
            bool sup = loCnt >= hiCnt
            SrZone z = SrZone.new(center, center + halfZ, center - halfZ, tch, bull, bear, sup, avgV)
            if tch >= minTouches
                array.push(zones, z)
    zones
