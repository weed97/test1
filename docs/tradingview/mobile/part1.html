213f_calcVP() =>
214 float hi = ta.highest(high, vpLookback)
215 float lo = ta.lowest(low, vpLookback)
216 float step = (hi - lo) / vpBins
217 float[] bins = array.new_float(vpBins, 0.0)
218 float totalVol = 0.0
219 for i = 0 to vpLookback - 1
220 if bar_index >= i
221 float c = close[i]
222 float v = volume[i]
223 int idx = step > 0 ? int(math.floor((c - lo) / step)) : 0
224 idx := math.min(math.max(idx, 0), vpBins - 1)
225 array.set(bins, idx, array.get(bins, idx) + v)
226 totalVol += v
227 int pocIdx = 0
228 float maxV = 0.0
229 for j = 0 to vpBins - 1
230 float bv = array.get(bins, j)
231 if bv > maxV
232 maxV := bv
233 pocIdx := j
234 float pocPrice = lo + step * (pocIdx + 0.5)
235 float vaVol = array.get(bins, pocIdx)
236 int loIdx = pocIdx
237 int hiIdx = pocIdx
238 float target = totalVol * valueAreaPct / 100.0
239 while vaVol < target and (loIdx > 0 or hiIdx < vpBins - 1)
240 float volBelow = loIdx > 0 ? array.get(bins, loIdx - 1) : 0.0
241 float volAbove = hiIdx < vpBins - 1 ? array.get(bins, hiIdx + 1) : 0.0
242 if volAbove >= volBelow and hiIdx < vpBins - 1
243 hiIdx += 1
244 vaVol += array.get(bins, hiIdx)
245 else if loIdx > 0
246 loIdx -= 1
247 vaVol += array.get(bins, loIdx)
248 else if hiIdx < vpBins - 1
249 hiIdx += 1
250 vaVol += array.get(bins, hiIdx)
251 else
252 break
253 float vah = lo + step * (hiIdx + 1)
254 float val = lo + step * loIdx
255 float pocStrength = totalVol > 0 ? maxV / totalVol * 100.0 : 0.0
256 int[] hvnIdx = array.new_int(3, -1)
257 for rank = 0 to 2
258 float best = 0.0
259 int bestI = -1
260 for j = 0 to vpBins - 1
261 if j != pocIdx
262 float bv = array.get(bins, j)
263 bool already = false
264 for r = 0 to rank - 1
265 if array.get(hvnIdx, r) == j

268 best := bv
269 bestI := j

272 for j = 0 to vpBins - 1
273 if j != pocIdx and array.get(bins, j) < maxV * 0.2
