88// 1-1 Session VWAP — 일일 세션 기준 VWAP
89var float sCumVol = 0.0
90var float sCumPV = 0.0
91var float sSumDev = 0.0
92var float sSumDev2 = 0.0
93var int sBars = 0
94
95newDay = ta.change(time("1D")) != 0
96if newDay
97 sCumVol := 0.0
98 sCumPV := 0.0
99 sSumDev := 0.0
100 sSumDev2 := 0.0
101 sBars := 0
102
103tp = hlc3
104sCumVol += volume
105sCumPV += tp * volume
106sessionVwap = sCumVol > 0 ? sCumPV / sCumVol : tp
107
108dev = close - sessionVwap
109sSumDev += dev
110sSumDev2 += dev * dev
111sBars += 1
112sessionStd = sBars > 1 ? math.sqrt(math.max((sSumDev2 - sSumDev * sSumDev / sBars) / (sBars - 1), 0.0)) : 0.0
113
114band1U = sessionVwap + sessionStd * vwapBandMult
115band2U = sessionVwap + sessionStd * 2.0 * vwapBandMult
116band3U = sessionVwap + sessionStd * 3.0 * vwapBandMult
117band1D = sessionVwap - sessionStd * vwapBandMult
118band2D = sessionVwap - sessionStd * 2.0 * vwapBandMult
119band3D = sessionVwap - sessionStd * 3.0 * vwapBandMult
120
121vwapColor = close >= sessionVwap ? color.new(color.green, 0) : color.new(color.red, 0)
122plot(sessionVwap, "Session VWAP", color=vwapColor, linewidth=2)
