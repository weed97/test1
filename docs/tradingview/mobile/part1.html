// 1-1 Session VWAP — 일일 세션 기준 VWAP
var float sCumVol  = 0.0
var float sCumPV   = 0.0
var float sSumDev  = 0.0
var float sSumDev2 = 0.0
var int   sBars    = 0

newDay = ta.change(time("1D")) != 0
if newDay
    sCumVol  := 0.0
    sCumPV   := 0.0
    sSumDev  := 0.0
    sSumDev2 := 0.0
    sBars    := 0

tp = hlc3
sCumVol  += volume
sCumPV   += tp * volume
sessionVwap = sCumVol > 0 ? sCumPV / sCumVol : tp

dev = close - sessionVwap
sSumDev  += dev
sSumDev2 += dev * dev
sBars    += 1
sessionStd = sBars > 1 ? math.sqrt(math.max((sSumDev2 - sSumDev * sSumDev / sBars) / (sBars - 1), 0.0)) : 0.0

band1U = sessionVwap + sessionStd * vwapBandMult
band2U = sessionVwap + sessionStd * 2.0 * vwapBandMult
band3U = sessionVwap + sessionStd * 3.0 * vwapBandMult
band1D = sessionVwap - sessionStd * vwapBandMult
band2D = sessionVwap - sessionStd * 2.0 * vwapBandMult
band3D = sessionVwap - sessionStd * 3.0 * vwapBandMult

vwapColor = close >= sessionVwap ? color.new(color.green, 0) : color.new(color.red, 0)
plot(sessionVwap, "Session VWAP", color=vwapColor, linewidth=2)

// 1-4 VWAP Zone touch analysis — VWAP 밴드 존 분석
f_zoneTouches(float level, float halfH, int lb) =>
    int touches = 0
    float touchVol = 0.0
    int bounces = 0
    bool broken = false
    for i = 1 to lb
        if bar_index >= i
            float zTop = level + halfH
            float zBot = level - halfH
            bool touched = high[i] >= zBot and low[i] <= zTop
            if touched
                touches += 1
                touchVol += volume[i]
                bool bouncedUp = false
                bool bouncedDn = false
                for j = 0 to 2
                    if i - j >= 0
                        if close[i - j] > zTop
                            bouncedUp := true
                        if close[i - j] < zBot
                            bouncedDn := true
                if bouncedUp or bouncedDn
                    bounces += 1
            // broken through
            if close[i] > zTop + halfH or close[i] < zBot - halfH
                broken := true
    float strength = touches > 0 ? math.min(100, int(touches * 15 + touchVol / math.max(volMa, 1) * 5)) : 0
    [touches, bounces, broken, strength]

vwapHalfH = atr14 * vwapZoneAtrMult
var box[] vwapZoneBoxes = array.new_box()
var label[] vwapZoneLabels = array.new_label()

f_drawVwapZone(float lvl, string name, bool isUpper) =>
    if showVwapZones and barstate.islast
        [tch, bnc, brk, str] = f_zoneTouches(lvl, vwapHalfH, vwapZoneLookback)
        isSupport = not isUpper and tch >= 2 and bnc >= 2 and not brk
        isResist  = isUpper and tch >= 2 and bnc >= 2 and not brk
        bgCol = brk ? color.new(color.gray, 75) :
             isSupport ? color.new(color.green, 80) :
             isResist  ? color.new(color.red, 80) :
             color.new(color.gray, 90)
        b = box.new(bar_index - vwapZoneLookback, lvl + vwapHalfH, bar_index + 20, lvl - vwapHalfH,
             border_color=color.new(color.gray, 60), bgcolor=bgCol, border_width=1)
        array.push(vwapZoneBoxes, b)
        lblTxt = name + "\nTouches: " + str.tostring(tch) + "\nStrength: " + str.tostring(str)
        l = label.new(bar_index + 5, lvl, lblTxt, style=label.style_label_left,
             color=color.new(color.black, 40), textcolor=color.white, size=size.tiny)
        array.push(vwapZoneLabels, l)

if barstate.islast
    for b in vwapZoneBoxes
        box.delete(b)
    for l in vwapZoneLabels
        label.delete(l)
    array.clear(vwapZoneBoxes)
    array.clear(vwapZoneLabels)
