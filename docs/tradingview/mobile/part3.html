// ══════════ SECTION 2: VOLUME PROFILE + POC ══════════
// 2-1 Rolling Volume Profile — 롤링 거래량 프로파일
f_calcVP() =>
    float hi = ta.highest(high, vpLookback)
    float lo = ta.lowest(low, vpLookback)
    float step = (hi - lo) / vpBins
    float[] bins = array.new_float(vpBins, 0.0)
    float totalVol = 0.0
    for i = 0 to vpLookback - 1
        if bar_index >= i
            float c = close[i]
            float v = volume[i]
            int idx = step > 0 ? int(math.floor((c - lo) / step)) : 0
            idx := math.min(math.max(idx, 0), vpBins - 1)
            array.set(bins, idx, array.get(bins, idx) + v)
            totalVol += v
    // POC
    int pocIdx = 0
    float maxV = 0.0
    for j = 0 to vpBins - 1
        float bv = array.get(bins, j)
        if bv > maxV
            maxV := bv
            pocIdx := j
    float pocPrice = lo + step * (pocIdx + 0.5)
    // Value Area
    float vaVol = array.get(bins, pocIdx)
    int loIdx = pocIdx
    int hiIdx = pocIdx
    float target = totalVol * valueAreaPct / 100.0
    while vaVol < target and (loIdx > 0 or hiIdx < vpBins - 1)
        float volBelow = loIdx > 0 ? array.get(bins, loIdx - 1) : 0.0
        float volAbove = hiIdx < vpBins - 1 ? array.get(bins, hiIdx + 1) : 0.0
        if volAbove >= volBelow and hiIdx < vpBins - 1
            hiIdx += 1
            vaVol += array.get(bins, hiIdx)
        else if loIdx > 0
            loIdx -= 1
            vaVol += array.get(bins, loIdx)
        else if hiIdx < vpBins - 1
            hiIdx += 1
            vaVol += array.get(bins, hiIdx)
        else
            break
    float vah = lo + step * (hiIdx + 1)
    float val = lo + step * loIdx
    float pocStrength = totalVol > 0 ? maxV / totalVol * 100.0 : 0.0
    // HVN top 3 (excluding POC)
    int[] hvnIdx = array.new_int(3, -1)
    for rank = 0 to 2
        float best = 0.0
        int bestI = -1
        for j = 0 to vpBins - 1
            if j != pocIdx
                float bv = array.get(bins, j)
                bool already = false
                for r = 0 to rank - 1
                    if array.get(hvnIdx, r) == j
                        already := true
                if not already and bv > best
                    best := bv
                    bestI := j
        array.set(hvnIdx, rank, bestI)
    // LVN indices
    int[] lvnIdx = array.new_int()
    for j = 0 to vpBins - 1
        if j != pocIdx and array.get(bins, j) < maxV * 0.2
            array.push(lvnIdx, j)
    [pocPrice, vah, val, pocStrength, step, lo, bins, hvnIdx, lvnIdx, pocIdx]

[poc, vah, val, pocStrength, binStep, binLo, vpBinsArr, hvnIndices, lvnIndices, pocBinIdx] = f_calcVP()

pocDistPct = poc > 0 ? (close - poc) / poc * 100.0 : 0.0
pocDistUsd = close - poc
