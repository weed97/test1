// ══════════ SECTION 3: BUY WALL + SELL WALL ══════════
// 3-1 ~ 3-4 Wall detection, zones, expiry — 매수·매도벽 탐지
type Wall
    float level
    int   barIdx
    float vol
    bool  isBuy
    bool  broken
    line  ln
    box   bx
    label lb

var Wall[] walls = array.new<Wall>()

f_nearWall(float price, float pct) =>
    bool near = false
    for w in walls
        if math.abs(w.level - price) / price * 100.0 < pct
            near := true
    near

f_strengthBar(float vol) =>
    float ratio = vol / math.max(volMa, 1)
    int blocks = int(math.min(8, math.round(ratio)))
    string s = ""
    for i = 0 to 7
        s += i < blocks ? "█" : "░"
    s

bigLowerWick = low < open - (high - low) * 0.5
bigUpperWick = high > open + (high - low) * 0.5
buyCond  = confirmed and volume > volMa * buyWallMult and (close >= open or bigLowerWick) and close < sessionVwap
sellCond = confirmed and volume > volMa * sellWallMult and (close <= open or bigUpperWick) and close > sessionVwap

if buyCond and not f_nearWall(low, 0.3)
    string bwStr = volume > volMa * strongWallMult ? "💪 STRONG BW" :
         volume > volMa * modWallMult ? "⚡ MODERATE BW" : "BW"
    string bar = f_strengthBar(volume)
    float bTop = low + atr14 * 0.5
    float bBot = low - atr14 * 0.2
    line  bln = line.new(bar_index, low, bar_index + wallExpiryBars, low, color=color.lime, width=2)
    box   bbx = box.new(bar_index, bTop, bar_index + wallExpiryBars, bBot,
         bgcolor=color.new(color.lime, 85), border_color=color.new(color.lime, 50))
    label blb = label.new(bar_index, bTop, bwStr + " " + bar, style=label.style_label_down,
         color=color.new(color.lime, 30), textcolor=color.white, size=size.small)
    Wall nw = Wall.new(low, bar_index, volume, true, false, bln, bbx, blb)
    array.push(walls, nw)
    if array.size(walls) > 5
        Wall old = array.shift(walls)
        line.delete(old.ln)
        box.delete(old.bx)
        label.delete(old.lb)

if sellCond and not f_nearWall(high, 0.3)
    string swStr = volume > volMa * strongWallMult ? "💪 STRONG SW" :
         volume > volMa * modWallMult ? "⚡ MODERATE SW" : "SW"
    string bar = f_strengthBar(volume)
    float sTop = high + atr14 * 0.2
    float sBot = high - atr14 * 0.5
    line  sln = line.new(bar_index, high, bar_index + wallExpiryBars, high, color=color.red, width=2)
    box   sbx = box.new(bar_index, sTop, bar_index + wallExpiryBars, sBot,
         bgcolor=color.new(color.red, 85), border_color=color.new(color.red, 50))
    label slb = label.new(bar_index, sBot, swStr + " " + bar, style=label.style_label_up,
         color=color.new(color.red, 30), textcolor=color.white, size=size.small)
    Wall nw = Wall.new(high, bar_index, volume, false, false, sln, sbx, slb)
    array.push(walls, nw)
    if array.size(walls) > 5
        Wall old = array.shift(walls)
        line.delete(old.ln)
        box.delete(old.bx)
        label.delete(old.lb)

// Wall expiry + break logic — 벽 만료·돌파 처리
var bool buyWallBroken  = false
var bool sellWallBroken = false
buyWallBroken  := false
sellWallBroken := false

int activeBuyWalls  = 0
int activeSellWalls = 0
float nearestBw = na
float nearestSw = na

if array.size(walls) > 0
    for i = array.size(walls) - 1 to 0
        Wall w = array.get(walls, i)
        bool expired = bar_index - w.barIdx > wallExpiryBars
        if expired
            line.delete(w.ln)
            box.delete(w.bx)
            label.delete(w.lb)
            array.remove(walls, i)
        else
            // extend drawings
            line.set_x2(w.ln, bar_index + 5)
            box.set_right(w.bx, bar_index + 5)
            if w.isBuy
                activeBuyWalls += 1
                if w.level <= close and (na(nearestBw) or w.level > nearestBw)
                    nearestBw := w.level
                if confirmed and close < w.level and not w.broken
                    w.broken := true
                    buyWallBroken := true
                    line.set_color(w.ln, color.gray)
                    label.set_text(w.lb, "BW BROKEN")
                    label.set_color(w.lb, color.new(color.gray, 30))
            else
                activeSellWalls += 1
                if w.level >= close and (na(nearestSw) or w.level < nearestSw)
                    nearestSw := w.level
                if confirmed and close > w.level and not w.broken
                    w.broken := true
                    sellWallBroken := true
                    line.set_color(w.ln, color.gray)
                    label.set_text(w.lb, "SW BROKEN")
                    label.set_color(w.lb, color.new(color.gray, 30))
            array.set(walls, i, w)

f_wallNear(float price, float pct, bool wantBuy) =>
    bool found = false
    for w in walls
        if w.isBuy == wantBuy and not w.broken and math.abs(w.level - price) / price * 100.0 < pct
            found := true
    found
