// ══════════ SECTION 5: WALL PRESSURE (무료·캔들 기반) ══════════
// 실제 호가는 Binance/Bybit 무료 API → tradingview_orderbook_feed.py
// Pine: 매수·매도벽 + S/R 태그만으로 Bid/Ask 압력 추정 (유료 API·랜덤 없음)

f_wallSizeAt(float price, float pct) =>
    float sz = 0.0
    for w in walls
        if not w.broken and math.abs(w.level - price) / price * 100.0 < pct
            sz += w.vol / math.max(volMa, 1) * 50.0
    sz

f_obTag(float price, bool isAsk) =>
    string tag = ""
    if isAsk
        if f_wallNear(price, 0.3, false)
            tag += "SW🧱 "
        if not na(nearMajorRes) and math.abs(price - nearMajorRes) / price * 100.0 < 0.3
            tag += "MR🔴 "
        if math.abs(price - band2U) / price * 100.0 < 0.3
            tag += "V+2σ "
    else
        if f_wallNear(price, 0.3, true)
            tag += "BW🧱 "
        if not na(nearMajorSup) and math.abs(price - nearMajorSup) / price * 100.0 < 0.3
            tag += "MS🟢 "
        if math.abs(price - poc) / price * 100.0 < 0.3
            tag += "POC⭐ "
        if math.abs(price - band1D) / price * 100.0 < 0.3
            tag += "V-1σ "
    tag

// 매 바 Bid/Ask 압력 (벽 기반만 — 테이블 꺼져도 대시보드·시그널 동작)
float bidTotal = 0.0
float askTotal = 0.0
float imbRatio = 1.0
float spacing = atr14 * obAtrSpacing
for n = 1 to obLevels
    askTotal += f_wallSizeAt(close + spacing * n, 0.25)
    bidTotal += f_wallSizeAt(close - spacing * n, 0.25)
for w in walls
    if not w.broken
        if w.isBuy
            bidTotal += w.vol / math.max(volMa, 1) * 10.0
        else
            askTotal += w.vol / math.max(volMa, 1) * 10.0
imbRatio := askTotal > 0 ? bidTotal / askTotal : 1.0

f_sizeBar(float sz, float maxSz) =>
    int blocks = int(math.round(sz / math.max(maxSz, 1) * 8))
    string s = ""
    for i = 0 to 7
        s += i < blocks ? "█" : "░"
    s
