5f_wallSizeAt(float price, float pct) =>
6 float sz = 0.0
7 for w in walls
8 if not w.broken and math.abs(w.level - price) / price * 100.0 < pct
9 sz += w.vol / math.max(volMa, 1) * 50.0
10 sz

32// 매 바 Bid/Ask 압력 (벽 기반만 — 테이블 꺼져도 대시보드·시그널 동작)
33float bidTotal = 0.0

36float spacing = atr14 * obAtrSpacing

38 askTotal += f_wallSizeAt(close + spacing * n, 0.25)

41 if not w.broken

44 else
45 askTotal += w.vol / math.max(volMa, 1) * 10.0
46imbRatio := askTotal > 0 ? bidTotal / askTotal : 1.0

48f_sizeBar(float sz, float maxSz) =>
49 int blocks = int(math.round(sz / math.max(maxSz, 1) * 8))

53 s
