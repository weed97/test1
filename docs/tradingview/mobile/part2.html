35f_zoneTouches(float level, float halfH, int lb) =>
36 int touches = 0
37 float touchVol = 0.0

39 bool broken = false
40 for i = 1 to lb
41 if bar_index >= i
42 float zTop = level + halfH

44 bool touched = high[i] >= zBot and low[i] <= zTop
45 if touched
46 touches += 1
47 touchVol += volume[i]
48 bool bouncedUp = false
49 bool bouncedDn = false
50 for j = 0 to 2

53 bouncedUp := true
54 if close[i - j] < zBot
55 bouncedDn := true
56 if bouncedUp or bouncedDn

58 // broken through
59 if close[i] > zTop + halfH or close[i] < zBot - halfH
60 broken := true
61 float strength = touches > 0 ? math.min(100, int(touches * 15 + touchVol / math.max(volMa, 1) * 5)) : 0
