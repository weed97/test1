f_buildZones() =>
    SrZone[] zones = array.new<SrZone>()
    float clusterR = atr14 * clusterAtrMult
    int pSize = array.size(pivotPrices)
    bool[] used = array.new_bool(pSize, false)
    for i = 0 to pSize - 1
        if not array.get(used, i)
            float pc = array.get(pivotPrices, i)
            bool isH = array.get(pivotIsHigh, i)
            float sumP = pc
            int cnt = 1
            int hiCnt = isH ? 1 : 0
            int loCnt = isH ? 0 : 1
            array.set(used, i, true)
            for j = i + 1 to pSize - 1
                if not array.get(used, j)
                    float pj = array.get(pivotPrices, j)
                    if math.abs(pj - pc) <= clusterR
                        sumP += pj
                        cnt += 1
                        if array.get(pivotIsHigh, j)
                            hiCnt += 1
                        else
                            loCnt += 1
                        array.set(used, j, true)
            float center = sumP / cnt
            float halfZ = atr14 * 0.3
            int tch = 0
            int bull = 0
            int bear = 0
            float tVol = 0.0
            for k = 0 to srLookback - 1
                if bar_index >= k
                    if high[k] >= center - halfZ and low[k] <= center + halfZ
                        tch += 1
                        tVol += volume[k]
                        bool bUp = false
                        bool bDn = false
                        for m = 0 to 2
                            if k - m >= 0
                                if close[k - m] > center + halfZ
                                    bUp := true
                                if close[k - m] < center - halfZ
                                    bDn := true
                        if bUp
                            bull += 1
                        if bDn
                            bear += 1
            float avgV = tch > 0 ? tVol / tch : 0.0
            bool sup = loCnt >= hiCnt
            SrZone z = SrZone.new(center, center + halfZ, center - halfZ, tch, bull, bear, sup, avgV)
            if tch >= minTouches
                array.push(zones, z)
    zones

f_wallSizeAt(float price, float pct) =>
    float sz = 0.0
    for w in walls
        if not w.broken and math.abs(w.level - price) / price * 100.0 < pct
            sz += w.vol / math.max(volMa, 1) * 50.0
    sz

f_obTag(float price, bool isAsk) =>
    string tag = ""
    if isAsk
        if f_wallNear(price, 0.3, false)
            tag += "SW🧱 "
        if not na(nearMajorRes) and math.abs(price - nearMajorRes) / price * 100.0 < 0.3
            tag += "MR🔴 "
        if math.abs(price - band2U) / price * 100.0 < 0.3
            tag += "V+2σ "
    else
        if f_wallNear(price, 0.3, true)
            tag += "BW🧱 "
        if not na(nearMajorSup) and math.abs(price - nearMajorSup) / price * 100.0 < 0.3
            tag += "MS🟢 "
        if math.abs(price - poc) / price * 100.0 < 0.3
            tag += "POC⭐ "
        if math.abs(price - band1D) / price * 100.0 < 0.3
            tag += "V-1σ "
    tag
